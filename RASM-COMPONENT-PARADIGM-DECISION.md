// The plastic/torsion/shear/warping/asymmetry supplement the elastic polygon integral cannot
// yield. The parametric closed forms fill the last four columns 0.0 (a doubly-symmetric solid or
// closed section places its shear centre AT the centroid); the relocated SteelStiffness algebra

public readonly record struct SectionSupplement(
    double Zx, double Zy, double J, double Avy, double Avz,
    double Iw = 0.0, double ShearCentreY = 0.0, double ShearCentreZ = 0.0, double Monosymmetry = 0.0);

// the closed profile axis — a new arm is compiler-forced HERE and nowhere else. Every arm routes
// the ONE VividOrange Green-theorem integral (`new SectionProperties((IProfile)perimeter)`)
// through the ONE twenty-column Admit lift, then the topology
// supplement: Forms closed forms for solid/hollow arms, the relocated SteelStiffness thin-walled
// algebra for open arms, the published-identity route for Catalogued. Layered/Nominal are
// unsectioned by design and fault loudly if a mis-flagged Sectioned row reaches them.
public static class SectionSolver {
    public static Fin<ComputedSection> Solve(SectionProfile profile, Op key) =>
        profile.Switch(
            rectangle:         r => Admit(Curves.Rect(r.WidthMm.Value, r.DepthMm.Value), r, Forms.SolidRectangle(r), key),

            roundedRectangle:  r => Admit(Curves.RoundedRect(r), r, Forms.SolidRectangle(new SectionProfile.Rectangle(r.WidthMm, r.DepthMm)), key),
            circle:            c => Admit(Curves.Disc(c), c, Forms.SolidCircle(c), key),
            circleHollow:      c => Admit(Curves.Ring(c), c, Forms.Pipe(c), key),
            ellipse:           e => Admit(Curves.Ellipse(e), e, Forms.SolidEllipse(e), key),
