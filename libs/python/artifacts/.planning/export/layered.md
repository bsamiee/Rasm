# [PY_ARTIFACTS_LAYERED]

The editable-export owner authoring named, separable layers for the Illustrator/InDesign/Photoshop hand-off — the highest-value export capability, named layers rather than one flattened path-soup splat. `LayeredExport` is ONE owner over the layered-emission pipeline carrying a closed `ExportOp` `expression.tagged_union` — a `SvgLayers` case (named `<g id=>` Groups), a `PdfOcg` case (PDF optional-content groups), a `PdfSurgery` case (low-level `/OCProperties` object-model surgery on an already-drawn PDF), and a `TiffLayers` case (a layered TIFF over Photoshop image-resource blocks) — each carrying its own typed `tuple[Layer, ...]` payload, never a `StrEnum` keyed against an erased `dict[str, object]` bag — dispatched by one total `match`+`assert_never`. It consumes the placed multi-source layout `composition/compose#COMPOSE` emits as one flat single-`<svg>`/single-PDF document (base graphic, registration overlay, n-up sheet) and the per-source named-layer source graphics `graphic/marks/encode#MARK` and `visualization/diagram/draw#DRAW` emit, and binds each as a NAMED editable layer rather than one flattened path soup. The `SvgLayers` arm wraps each placed source in a `drawsvg` `Group(id=name)` and serializes the nested named-group document through `Drawing.as_svg` on the cp315 core; the `PdfOcg` arm authors PDF optional-content groups through `pymupdf` `Document.add_ocg(name, intent, usage) -> xref` and binds each placed source's drawn content to its layer with the `oc=<xref>` content-binding parameter on `Page.show_pdf_page`/`insert_image`/`insert_text`/`insert_htmlbox`/`draw_*`, in-process on the cp315 core because pymupdf is the `cp310-abi3` stable-ABI wheel that reflects on cp315 ungated; the `PdfSurgery` arm is the low-level alternative that mutates `/Root/OCProperties` directly through the `pikepdf` qpdf-native object model when the layer structure must be grafted onto an already-drawn PDF the pymupdf draw-binding cannot reach; the `TiffLayers` arm assembles a Photoshop-compatible layered TIFF over `psdtags`+`tifffile`. One export surface discriminating the target format and the layer engine, never a per-producer layer-emit arm and never a per-format writer family. The placement, scaling, and rasterization are upstream concerns: this owner re-renders nothing and re-lays-out nothing — it receives an already-placed flat layout plus the per-region source split and authors the layer structure the flat egress deliberately omits, so a base-graphic/registration-overlay/n-up-sheet artifact lands once as named layers an external editor can toggle, lock, and re-color. ```python signature
from collections.abc import Sequence
from enum import StrEnum
from io import BytesIO
from typing import Literal, assert_never

from expression import case, tag, tagged_union
from msgspec import Struct

from rasm.runtime.content_identity import ContentIdentity, ContentKey
from rasm.runtime.faults import RuntimeRail, async_boundary

from artifacts.core.receipt import ArtifactReceipt


class LayerView(StrEnum):
    VIEW = "View"
    PRINT = "Print"
    EXPORT = "Export"


class Layer(Struct, frozen=True):
    name: str
    source: bytes
    bbox: tuple[float, float, float, float]
    visible: bool = True
    locked: bool = False


@tagged_union(frozen=True)
class ExportOp:
    tag: Literal["svg_layers", "pdf_ocg", "pdf_surgery", "tiff_layers"] = tag()
    svg_layers: tuple[Layer, ...] = case()
    pdf_ocg: tuple[tuple[Layer, ...], bytes, LayerView] = case()
    pdf_surgery: tuple[tuple[Layer, ...], bytes] = case()
    tiff_layers: tuple[Layer, ...] = case()

    @staticmethod
    def SvgLayers(layers: tuple[Layer, ...]) -> "ExportOp":
        return ExportOp(svg_layers=layers)

    @staticmethod
    def PdfOcg(layers: tuple[Layer, ...], base: bytes, view: LayerView = LayerView.VIEW) -> "ExportOp":
        return ExportOp(pdf_ocg=(layers, base, view))

    @staticmethod
    def PdfSurgery(layers: tuple[Layer, ...], base: bytes) -> "ExportOp":
        return ExportOp(pdf_surgery=(layers, base))

    @staticmethod
    def TiffLayers(layers: tuple[Layer, ...]) -> "ExportOp":
        return ExportOp(tiff_layers=layers)


class LayeredExport(Struct, frozen=True):
    op: ExportOp

    async def of(self) -> RuntimeRail[ContentKey]:
        return await async_boundary(f"export.{self.op.tag}", self._emit)

    async def _emit(self) -> ContentKey:
        match self.op:
            case ExportOp(tag="svg_layers", svg_layers=layers):
                data = _svg_layers(layers)
            case ExportOp(tag="pdf_ocg", pdf_ocg=(layers, base, view)):
                data = _pdf_ocg(layers, base, view.value)
            case ExportOp(tag="pdf_surgery", pdf_surgery=(layers, base)):
                data = _pdf_surgery(layers, base)
            case ExportOp(tag="tiff_layers", tiff_layers=layers):
                data = _tiff_layers(layers)
            case _:
                assert_never(self.op)
        return ContentIdentity.of(f"export-{self.op.tag}", data)


def _viewport(layers: Sequence[Layer]) -> tuple[float, float]:
    return (
        max((layer.bbox[2] for layer in layers), default=0.0),
        max((layer.bbox[3] for layer in layers), default=0.0),
    )


def _svg_layers(layers: tuple[Layer, ...]) -> bytes:
    import drawsvg

    width, height = _viewport(layers)
    drawing = drawsvg.Drawing(width, height, origin=(0.0, 0.0))
    for layer in layers:
        group = drawsvg.Group(id=layer.name, style=None if layer.visible else "display:none")
        group.append(drawsvg.Raw(layer.source.decode()))
        drawing.append(group)
    return drawing.as_svg().encode()


def _pdf_ocg(layers: tuple[Layer, ...], base: bytes, view: str) -> bytes:
    import pymupdf

    doc = pymupdf.open(stream=base, filetype="pdf")
    page = doc[0]
    on, off, locked = [], [], []
    for layer in layers:
        xref = doc.add_ocg(layer.name, on=layer.visible, intent=view, usage="Artwork")
        (on if layer.visible else off).append(xref)
        if layer.locked:
            locked.append(xref)
        src = pymupdf.open(stream=layer.source, filetype="pdf")
        page.show_pdf_page(pymupdf.Rect(layer.bbox), src, 0, oc=xref)
    doc.set_layer(0, on=on, off=off, locked=locked)
    return doc.tobytes(garbage=3, deflate=True)


def _pdf_surgery(layers: tuple[Layer, ...], base: bytes) -> bytes:
    import pikepdf
    from pikepdf import Array, Dictionary, Name

    pdf = pikepdf.open(BytesIO(base))
    ocgs = [pdf.make_indirect(Dictionary(Type=Name.OCG, Name=layer.name)) for layer in layers]
    visible = [ocg for layer, ocg in zip(layers, ocgs, strict=True) if layer.visible]
    hidden = [ocg for layer, ocg in zip(layers, ocgs, strict=True) if not layer.visible]
    pdf.Root.OCProperties = Dictionary(
        OCGs=Array(ocgs),
        D=Dictionary(Order=Array(ocgs), ON=Array(visible), OFF=Array(hidden)),
    )
    page = pdf.pages[0]
    spans = [b"/OC " + str(page.add_resource(ocg, Name.Properties)).encode() + b" BDC\nEMC\n" for ocg in ocgs]
    page.contents_add(pdf.make_stream(b"".join(spans)))
    sink = BytesIO()
    pdf.save(sink)
    return sink.getvalue()


def _tiff_layers(layers: tuple[Layer, ...]) -> bytes:  # [RESEARCH] psdtags + tifffile not yet admitted
    import tifffile
    from psdtags import PsdLayers, TiffImageSourceData

    sink = BytesIO()
    blocks = TiffImageSourceData(PsdLayers([_psd_layer(layer) for layer in layers]))
    tifffile.imwrite(sink, _composite(layers), extratags=[blocks.tifftag()])
    return sink.getvalue()
